# TC512 — REQ-002 Write to read-only namespace
import pytest

def test_write_readonly_ns_rejected(ctrl):
    """Writing to a read-only NS must fail with 0x82."""
    ns = ctrl.get_namespace(nsid=2, readonly=True)
    r = ctrl.write(nsid=ns.id, lba=0x0, data=b"x")
    assert r.status == 0x82  # Namespace is write protected
